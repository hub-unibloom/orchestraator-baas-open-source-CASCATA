
map $http_upgrade $connection_upgrade {
    default upgrade;
    '' close;
}

# Definição de Zonas de Rate Limit
limit_req_zone $binary_remote_addr zone=api_limit:10m rate=1000r/s;
limit_req_zone $binary_remote_addr zone=auth_limit:10m rate=100r/s;
limit_conn_zone $binary_remote_addr zone=conn_limit:10m;

# ============================================================
# CASCATA ZERO-NETWORK HOT-PATH (Hyper-Cluster)
# Upstream balanceia a carga usando Unix Domain Sockets 
# direto na memória RAM, ignorando a pilha TCP/IP do Linux.
# ============================================================
upstream backend_data_sockets {
    # Suportamos até 4 workers Socket por padrão. O NGINX ignora os que estiverem down.
    server unix:/tmp/cascata_sockets/worker_1.sock max_fails=3 fail_timeout=5s;
    server unix:/tmp/cascata_sockets/worker_2.sock max_fails=3 fail_timeout=5s;
    server unix:/tmp/cascata_sockets/worker_3.sock max_fails=3 fail_timeout=5s;
    server unix:/tmp/cascata_sockets/worker_4.sock max_fails=3 fail_timeout=5s;
    
    # Fallback de segurança para porta TCP caso sockets falhem
    server backend_data:3000 backup;
}

# --- BLOCO PADRÃO HTTP (Porta 80) ---
server {
    listen 80 default_server;
    server_name _;
    
    server_tokens off;
    
    # HARDENING: Increased to 5GB for Full Database Imports/Exports
    client_max_body_size 5G;

    location /.well-known/acme-challenge/ {
        root /var/www/html;
        allow all;
    }

    # Lógica de Isolamento Zero Trust
    location / {
        if ($host !~* ^(localhost|127\.0\.0\.1|172\.|10\.|[0-9]+\.[0-9]+\.[0-9]+\.[0-9]+)$ ) {
            return 444; 
        }

        limit_req zone=api_limit burst=20 nodelay;
        limit_conn conn_limit 20;
        
        proxy_pass http://frontend;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    # API Control Plane (Long Timeout for Restores)
    location /api/control/ {
        if ($host !~* ^(localhost|127\.0\.0\.1|172\.|10\.|[0-9]+\.[0-9]+\.[0-9]+\.[0-9]+)$ ) {
            return 444;
        }
        limit_req zone=api_limit burst=10 nodelay;
        # Control Plane permanece na porta TCP padrão (não focado em hiper-rendimento, mas estabilidade)
        proxy_pass http://backend_control:3000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        
        # Extended timeouts for Import/Export operations
        proxy_read_timeout 3600s;
        proxy_send_timeout 3600s;
    }

    # API Data Plane (Hyper-Cluster Via Unix Sockets)
    location ~ ^/(api/data/|rpc/|auth/|storage/|edge/|tables/|rest/|vector/) {
        # O Pulo do Gato: Não podemos bloquear $host diferente de IP aqui,
        # pois tenants usam custom domains (ex: Cloudflare). O bloqueio de
        # segurança Zero-Trust fica no Control Plane e Painel de Admin.
        
        limit_req zone=api_limit burst=50 nodelay;
        limit_conn conn_limit 50;
        
        # O Pulo do Gato: NGINX entrega direto na RAM via Socket
        proxy_pass http://backend_data_sockets;
        
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        
        proxy_buffering off;
        proxy_cache off;
        proxy_read_timeout 86400s;
        proxy_send_timeout 86400s;
    }
}

# --- BLOCO PADRÃO HTTPS (Porta 443) ---
server {
    listen 443 ssl default_server;
    server_name _;
    
    server_tokens off;
    client_max_body_size 5G; # Match HTTP limit

    ssl_certificate /etc/letsencrypt/live/system/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/system/privkey.pem;

    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers HIGH:!aNULL:!MD5;
    ssl_prefer_server_ciphers on;

    location / {
        return 444;
    }
}

include /etc/nginx/conf.d/dynamic/*.conf;
